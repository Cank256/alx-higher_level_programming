# Test cases for MyList class

# Test Case 1: Create a MyList and print it sorted
Create a MyList with elements [3, 1, 2, 5, 4] and print it sorted.
Expected Output:
1 2 3 4 5

# Test Case 2: Create an empty MyList and print it sorted
Create an empty MyList and print it sorted.
Expected Output: (Empty list, nothing to print)

# Test Case 3: Create a MyList with a single element and print it sorted
Create a MyList with a single element [42] and print it sorted.
Expected Output:
42

# Test Case 4: Create a MyList with negative integers and print it sorted
Create a MyList with elements [-5, -1, -3, -2, -4] and print it sorted.
Expected Output:
-5 -4 -3 -2 -1

# Test Case 5: Create a MyList with repeated elements and print it sorted
Create a MyList with elements [1, 2, 1, 3, 2, 3] and print it sorted.
Expected Output:
1 1 2 2 3 3
